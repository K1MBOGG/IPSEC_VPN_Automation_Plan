# Palo Alto example configuration - conceptual

set network ike crypto-profiles ike-crypto-profiles vpn-ike dh-group group14
set network ike crypto-profiles ike-crypto-profiles vpn-ike encryption aes-256-cbc
set network ike crypto-profiles ike-crypto-profiles vpn-ike hash sha256

set network ike gateway vpn-to-fortigate authentication pre-shared-key key vpn_shared_key
set network ike gateway vpn-to-fortigate protocol ikev2 ike-crypto-profile vpn-ike
set network ike gateway vpn-to-fortigate local-address ip 198.51.100.20
set network ike gateway vpn-to-fortigate peer-address ip 203.0.113.10
set network ike gateway vpn-to-fortigate interface ethernet1/1

set network tunnel ipsec crypto-profiles ipsec-crypto-profiles vpn-ipsec esp authentication sha256
set network tunnel ipsec crypto-profiles ipsec-crypto-profiles vpn-ipsec esp encryption aes-256-cbc
set network tunnel ipsec crypto-profiles ipsec-crypto-profiles vpn-ipsec dh-group group14

set network interface tunnel units tunnel.1 ip 169.254.1.2/30
set network interface tunnel units tunnel.1 mtu 1420

set network tunnel ipsec vpn vpn-to-fortigate tunnel-interface tunnel.1
set network tunnel ipsec vpn vpn-to-fortigate auto-key ike-gateway vpn-to-fortigate
set network tunnel ipsec vpn vpn-to-fortigate auto-key ipsec-crypto-profile vpn-ipsec

set network virtual-router default routing-table ip static-route route-to-fg destination 10.10.10.0/24 interface tunnel.1

set rulebase security rules allow-vpn from trust to vpn source any destination any application any service application-default action allow
set rulebase nat rules no-nat-vpn from trust to vpn source 192.168.1.0/24 destination 10.10.10.0/24 service any source-translation off
